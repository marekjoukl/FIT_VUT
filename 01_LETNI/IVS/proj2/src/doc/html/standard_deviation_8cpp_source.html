20/**
21 * @brief Calculate the mean of the numbers in a vector
22 * @param values A vector of double values
23 * @return The mean value of the input vector
24 */
25double Mean(const std::vector<double> &values)
26{
27 double sum = 0.0;
28 for (double value : values)
29 {
30 sum = Addition(sum, value);
31 }
32
33 double mean = Division(sum, values.size());
34 return mean;
35}

37/**

39 * @param values A vector of double values

43double SumOfSquares(const std::vector<double> &values, double mean)
44{

55/**
56 * @brief Calculates the variance of the provided vector.
57 * @param values A vector of double values
58 * @return The variance of the input vector
59 */
60double Variance(const std::vector<double> &values)
61{
62 if (values.empty())
63 throw std::invalid_argument("Input vector is empty");
64
65 double meanValue = Mean(values);
66 double sumOfSquaresValue = SumOfSquares(values, meanValue);
67 double variance = Division(sumOfSquaresValue, values.size());
68 return variance;
69}
